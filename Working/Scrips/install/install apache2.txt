# Install Apache
sudo apt-get install apache2

# Check apache status
systemctl status apache2

# Restart apache
sudo systemctl restart apache2

# Create config file
sudo cp /etc/apache2/sites-available/000-default.conf /etc/apache2/sites-available/sohebox.conf

# Update config file
sudo nano /etc/apache2/sites-available/sohebox.conf

        ServerAdmin hientranduy@gmail.com
        ServerName sohebox.com
        ServerAlias www.sohebox.com     
        DocumentRoot /var/www/sohebox

        <Directory /var/www/sohebox>
                Options -Indexes +FollowSymLinks
                AllowOverride All
        </Directory>

        ErrorLog ${APACHE_LOG_DIR}/error.log
        CustomLog ${APACHE_LOG_DIR}/access.log combined

# Applu config 
sudo a2ensite sohebox

# Create check config/port status
sudo apachectl -S


# Deploy source
Create folder 
	sudo mdir /var/www/sohebox

Copy source website to this folder


# Give permission 
sudo chmod -R 755 /var/www



=================== Set up SSL ============================

# Install certbot
sudo apt install certbot python3-certbot-apache

		
# Generate certificate 
sudo certbot --apache
        ServerName sohebox.com
        ServerAlias www.sohebox.com  
		
	/etc/letsencrypt/live/sohebox.com/fullchain.pem
	/etc/letsencrypt/live/sohebox.com/privkey.pem		
	This certificate expires on 2022-09-23.

# Auto renew
sudo systemctl status certbot.timer


# Disable filewall
sudo ufw disable
